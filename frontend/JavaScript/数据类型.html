# еј•з”Ёзұ»еһӢ

  • Object
  • Function

let func = new Function ([arg1[, arg2[, ...g1N],] functionBody) жҚўеҸҘиҜқиҜҙпјҢеҮҪж•°зҡ„еҸӮж•°(жҲ–жӣҙзЎ®еҲҮең°иҜҙпјҢеҗ„еҸӮж•°зҡ„еҗҚз§°)йҰ–е…ҲеҮәзҺ°пјҢиҖҢеҮҪж•°дҪ“еңЁжңҖеҗҺгҖӮжүҖжңүеҸӮж•°йғҪеҶҷжҲҗеӯ—з¬ҰдёІеҪўејҸгҖӮйҖҡиҝҮжҹҘзңӢзӨәдҫӢпјҢеҸҜд»Ҙжӣҙе®№жҳ“зҗҶи§ЈгҖӮиҝҷжҳҜдёҖдёӘжңүдёӨдёӘеҸӮж•°зҡ„еҮҪж•°: let sum = new Function('a', 'b', 'returna + b'); alert( sum(1,2)); // 3

дҪҶжҳҜдёҖиҲ¬дёҚиҝҷд№ҲеҶҷпјҢеҸӘжҳҜжғіи®°еҪ•иҝҳжңүиҝҷдёҖз§Қе®ҡд№үеҮҪж•°зҡ„ж–№жі•

# зӣёе…ій—®йўҳ

  1. undefinedе’Ңnullзҡ„еҢәеҲ«
    • undefinedиЎЁзӨәе®ҡд№үдәҶжңӘиөӢеҖјпјҢеҚіundefinedиЎЁзӨә"зјәе°‘еҖј"пјҢе°ұжҳҜжӯӨеӨ„еә”иҜҘжңүдёҖдёӘеҖј
    • nullиЎЁзӨәе®ҡд№үе№¶иөӢеҖјдәҶпјҢеҸӘжҳҜеҖјдёәnullпјҢеҚіnullиЎЁзӨә"жІЎжңүеҜ№иұЎ"пјҢеҚіиҜҘеӨ„дёҚеә”иҜҘжңүеҖј
    • JavaScriptзҡ„жңҖеҲқзүҲжң¬жҳҜиҝҷж ·еҢәеҲҶзҡ„пјҡnullжҳҜдёҖдёӘиЎЁзӨә"ж— "зҡ„еҜ№иұЎпјҢиҪ¬дёәж•°еҖјж—¶дёә0пјӣundefinedжҳҜдёҖдёӘиЎЁзӨә"ж— "зҡ„еҺҹе§ӢеҖјпјҢиҪ¬дёәж•°еҖјж—¶дёәNaN

зӯүдәҺundefinedзҡ„жғ…еҶөпјҡ

  1. еҸҳйҮҸиў«еЈ°жҳҺдәҶпјҢдҪҶжІЎжңүиөӢеҖјж—¶пјҢе°ұзӯүдәҺundefined

  2. и°ғз”ЁеҮҪж•°ж—¶пјҢеә”иҜҘжҸҗдҫӣзҡ„еҸӮж•°жІЎжңүжҸҗдҫӣпјҢиҜҘеҸӮж•°зӯүдәҺundefined

  3. еҜ№иұЎжІЎжңүиөӢеҖјзҡ„еұһжҖ§пјҢиҜҘеұһжҖ§зҡ„еҖјдёәundefined

  4. еҮҪж•°жІЎжңүиҝ”еӣһеҖјж—¶пјҢй»ҳи®Өиҝ”еӣһundefinedгҖӮ

зӯүдәҺnullзҡ„жғ…еҶөпјҡ

  1. дҪңдёәеҮҪж•°зҡ„еҸӮж•°пјҢиЎЁзӨәиҜҘеҮҪж•°зҡ„еҸӮж•°дёҚжҳҜеҜ№иұЎ

  2. дҪңдёәеҜ№иұЎеҺҹеһӢй“ҫзҡ„з»ҲзӮ№

  1. д»Җд№Ҳж—¶еҖҷиөӢеҖјдёәnull
    • еҲқе§ӢиөӢеҖјпјҢиЎЁжҳҺе°ҶиҰҒиөӢеҖјдёәеҜ№иұЎ
    • з»“жқҹеүҚпјҢи®©еҜ№иұЎжҲҗдёәеһғеңҫеҜ№иұЎ(иў«еһғеңҫеӣһж”¶еҷЁеӣһж”¶)
